00122         void Step(Settings* settings)
00123         {
00124                 Test::Step(settings);
00125 
00126                 // We are going to destroy some bodies according to contact
00127                 // points. We must buffer the bodies that should be destroyed
00128                 // because they may belong to multiple contact points.
00129                 const int32 k_maxNuke = 6;
00130                 b2Body* nuke[k_maxNuke];
00131                 int32 nukeCount = 0;
00132 
00133                 // Traverse the contact results. Destroy bodies that
00134                 // are touching heavier bodies.
00135                 for (int32 i = 0; i < m_pointCount; ++i)
00136                 {
00137                         ContactPoint* point = m_points + i;
00138 
00139                         b2Body* body1 = point->fixtureA->GetBody();
00140                         b2Body* body2 = point->fixtureB->GetBody();
00141                         float32 mass1 = body1->GetMass();
00142                         float32 mass2 = body2->GetMass();
00143 
00144                         if (mass1 > 0.0f && mass2 > 0.0f)
00145                         {
00146                                 if (mass2 > mass1)
00147                                 {
00148                                         nuke[nukeCount++] = body1;
00149                                 }
00150                                 else
00151                                 {
00152                                         nuke[nukeCount++] = body2;
00153                                 }
00154 
00155                                 if (nukeCount == k_maxNuke)
00156                                 {
00157                                         break;
00158                                 }
00159                         }
00160                 }
00161 
00162                 // Sort the nuke array to group duplicates.
00163                 std::sort(nuke, nuke + nukeCount);
00164 
00165                 // Destroy the bodies, skipping duplicates.
00166                 int32 i = 0;
00167                 while (i < nukeCount)
00168                 {
00169                         b2Body* b = nuke[i++];
00170                         while (i < nukeCount && nuke[i] == b)
00171                         {
00172                                 ++i;
00173                         }
00174 
00175                         if (b != m_bomb)
00176                         {
00177                                 m_world->DestroyBody(b);
00178                         }
00179                 }
00180         }
