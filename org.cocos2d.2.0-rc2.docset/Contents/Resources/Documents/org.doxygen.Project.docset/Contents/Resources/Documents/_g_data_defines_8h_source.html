00069 // GDATA_ASSERT is like NSAssert, but takes a variable number of arguments:
00070 //
00071 //     GDATA_ASSERT(condition, @"Problem in argument %@", argStr);
00072 //
00073 // GDATA_DEBUG_ASSERT is similar, but compiles in only for debug builds

00076 #ifndef GDATA_ASSERT
00077   // we directly invoke the NSAssert handler so we can pass on the varargs
00078   #if !defined(NS_BLOCK_ASSERTIONS)
00079     #define GDATA_ASSERT(condition, ...)                                 \
00080       do {                                                                   \
00081         if (!(condition)) {                                                  \
00082           [[NSAssertionHandler currentHandler]                               \
00083               handleFailureInFunction:[NSString stringWithCString:__PRETTY_FUNCTION__] \
00084                                  file:[NSString stringWithCString:__FILE__]  \
00085                            lineNumber:__LINE__                               \
00086                           description:__VA_ARGS__];                          \
00087         }                                                                    \
00088       } while(0)
00089   #else
00090     #define GDATA_ASSERT(condition, ...) do { } while (0)
00091   #endif // !defined(NS_BLOCK_ASSERTIONS)
00092 #endif // GDATA_ASSERT
