\begin{DoxyCode}
                      :(ccTime) t scene:(CCScene*)s
{
        NSAssert( s != nil, @"Argument scene must be non-nil");

        if( (self=[super init]) ) {

                duration_ = t;

                // retain
                inScene_ = [s retain];
                outScene_ = [[CCDirector sharedDirector] runningScene];
                [outScene_ retain];

                NSAssert( inScene_ != outScene_, @"Incoming scene must be
       different from the outgoing scene" );

                [self sceneOrder];
        }
        return self;
}
