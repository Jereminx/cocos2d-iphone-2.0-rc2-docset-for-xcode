00557 /*
00558  * A call stack can be specified to the JS engine such that all JS entry/exits
00559  * to functions push/pop an entry to/from the specified stack.
00560  *
00561  * For more detailed information, see vm/SPSProfiler.h
00562  */
00563 class ProfileEntry
00564 {
00565     /*
00566      * All fields are marked volatile to prevent the compiler from re-ordering
00567      * instructions. Namely this sequence:
00568      *
00569      *    entry[size] = ...;
00570      *    size++;
00571      *
00572      * If the size modification were somehow reordered before the stores, then
00573      * if a sample were taken it would be examining bogus information.
00574      *
00575      * A ProfileEntry represents both a C++ profile entry and a JS one. Both use
00576      * the string as a description, but JS uses the sp as NULL to indicate that
00577      * it is a JS entry. The script_ is then only ever examined for a JS entry,
00578      * and the idx is used by both, but with different meanings.
00579      */
00580     const char * volatile string; // Descriptive string of this entry
00581     void * volatile sp;           // Relevant stack pointer for the entry
00582     JSScript * volatile script_;  // if js(), non-null script which is running
00583     int32_t volatile idx;         // if js(), idx of pc, otherwise line number
00584 
00585   public:
00586     /*
00587      * All of these methods are marked with the 'volatile' keyword because SPS's
00588      * representation of the stack is stored such that all ProfileEntry
00589      * instances are volatile. These methods would not be available unless they
00590      * were marked as volatile as well
00591      */
00592 
00593     bool js() volatile {
00594         JS_ASSERT_IF(sp == NULL, script_ != NULL);
00595         return sp == NULL;
00596     }
00597 
00598     uint32_t line() volatile { JS_ASSERT(!js()); return idx; }
00599     JSScript *script() volatile { JS_ASSERT(js()); return script_; }
00600     void *stackAddress() volatile { return sp; }
00601     const char *label() volatile { return string; }
00602 
00603     void setLine(uint32_t line) volatile { JS_ASSERT(!js()); idx = line; }
00604     void setLabel(const char *string) volatile { this->string = string; }
00605     void setStackAddress(void *sp) volatile { this->sp = sp; }
00606     void setScript(JSScript *script) volatile { script_ = script; }
00607 
00608     /* we can't know the layout of JSScript, so look in vm/SPSProfiler.cpp */
00609     JS_FRIEND_API(jsbytecode *) pc() volatile;
00610     JS_FRIEND_API(void) setPC(jsbytecode *pc) volatile;
