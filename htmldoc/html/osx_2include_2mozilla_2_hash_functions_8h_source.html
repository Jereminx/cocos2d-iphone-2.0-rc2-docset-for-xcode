00008 /*
00009  * This file exports functions for hashing data down to a 32-bit value,
00010  * including:
00011  *
00012  *  - HashString    Hash a char* or uint16_t/wchar_t* of known or unknown
00013  *                  length.
00014  *
00015  *  - HashBytes     Hash a byte array of known length.
00016  *
00017  *  - HashGeneric   Hash one or more values.  Currently, we support uint32_t,
00018  *                  types which can be implicitly cast to uint32_t, data
00019  *                  pointers, and function pointers.
00020  *
00021  *  - AddToHash     Add one or more values to the given hash.  This supports the
00022  *                  same list of types as HashGeneric.
00023  *
00024  *
00025  * You can chain these functions together to hash complex objects.  For example:
00026  *
00027  *  class ComplexObject
00028  *  {
00029  *      char* str;
00030  *      uint32_t uint1, uint2;
00031  *      void (*callbackFn)();
00032  *
00033  *    public:
00034  *      uint32_t hash() {
00035  *        uint32_t hash = HashString(str);
00036  *        hash = AddToHash(hash, uint1, uint2);
00037  *        return AddToHash(hash, callbackFn);
00038  *      }
00039  *  };
00040  *
00041  * If you want to hash an nsAString or nsACString, use the HashString functions
00042  * in nsHashKey.h.
00043  */

00061 inline uint32_t
00062 RotateBitsLeft32(uint32_t value, uint8_t bits)
00063 {
00064   MOZ_ASSERT(bits < 32);
00065   return (value << bits) | (value >> (32 - bits));

00070 inline uint32_t
00071 AddU32ToHash(uint32_t hash, uint32_t value)
00072 {
00073   /*
00074    * This is the meat of all our hash routines.  This hash function is not
00075    * particularly sophisticated, but it seems to work well for our mostly
00076    * plain-text inputs.  Implementation notes follow.
00077    *
00078    * Our use of the golden ratio here is arbitrary; we could pick almost any
00079    * number which:
00080    *
00081    *  * is odd (because otherwise, all our hash values will be even)
00082    *
00083    *  * has a reasonably-even mix of 1's and 0's (consider the extreme case
00084    *    where we multiply by 0x3 or 0xeffffff -- this will not produce good
00085    *    mixing across all bits of the hash).
00086    *
00087    * The rotation length of 5 is also arbitrary, although an odd number is again
00088    * preferable so our hash explores the whole universe of possible rotations.
00089    *
00090    * Finally, we multiply by the golden ratio *after* xor'ing, not before.
00091    * Otherwise, if |hash| is 0 (as it often is for the beginning of a message),
00092    * the expression
00093    *
00094    *   (GoldenRatioU32 * RotateBitsLeft(hash, 5)) |xor| value
00095    *
00096    * evaluates to |value|.
00097    *
00098    * (Number-theoretic aside: Because any odd number |m| is relatively prime to
00099    * our modulus (2^32), the list
00100    *
00101    *    [x * m (mod 2^32) for 0 <= x < 2^32]
00102    *
00103    * has no duplicate elements.  This means that multiplying by |m| does not
00104    * cause us to skip any possible hash values.
00105    *
00106    * It's also nice if |m| has large-ish order mod 2^32 -- that is, if the
00107    * smallest k such that m^k == 1 (mod 2^32) is large -- so we can safely
00108    * multiply our hash value by |m| a few times without negating the
00109    * multiplicative effect.  Our golden ratio constant has order 2^29, which is
00110    * more than enough for our purposes.)
00111    */
00112   return GoldenRatioU32 * (RotateBitsLeft32(hash, 5) ^ value);
00113 }

00129 template<>
00130 inline uint32_t
00131 AddUintptrToHash<8>(uint32_t hash, uintptr_t value)
00132 {
00133   /*
00134    * The static cast to uint64_t below is necessary because this function
00135    * sometimes gets compiled on 32-bit platforms (yes, even though it's a
00136    * template and we never call this particular override in a 32-bit build).  If
00137    * we do value >> 32 on a 32-bit machine, we're shifting a 32-bit uintptr_t
00138    * right 32 bits, and the compiler throws an error.
00139    */
00140   uint32_t v1 = static_cast<uint32_t>(value);
00141   uint32_t v2 = static_cast<uint32_t>(static_cast<uint64_t>(value) >> 32);
00142   return AddU32ToHash(AddU32ToHash(hash, v1), v2);
00143 }
