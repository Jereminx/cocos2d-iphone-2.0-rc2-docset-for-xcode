- (void) update: (ccTime)  dt

triggers the timer 

              : (ccTime) dt
{
        if( _elapsed == - 1)
        {
                _elapsed = 0;
                _nTimesExecuted = 0;
        }
        else
        {
                if (_runForever && !_useDelay)
                {//standard timer usage
                        _elapsed += dt;
                        if( _elapsed >= _interval ) {
                                [self trigger];
                                _elapsed = 0;

                        }
                }
                else
                {//advanced usage
                        _elapsed += dt;
                        if (_useDelay)
                        {
                                if( _elapsed >= _delay )
                                {
                                        [self trigger];
                                        _elapsed = _elapsed - _delay;
                                        _nTimesExecuted+=1;
                                        _useDelay = NO;
                                }
                        }
                        else
                        {
                                if (_elapsed >= _interval)
                                {
                                        [self trigger];
                                        _elapsed = 0;
                                        _nTimesExecuted += 1;

                                }
                        }

                        if (!_runForever && _nTimesExecuted > _repeat)
                        {       //unschedule timer
                                [self cancel];
                        }
                }
        }
}
