00020 /*
00021  * A wrapper is essentially a proxy that restricts access to certain traps. The
00022  * way in which a wrapper restricts access to its traps depends on the
00023  * particular policy for that wrapper. To allow a wrapper's policy to be
00024  * customized, the Wrapper base class contains two functions, enter/leave, which
00025  * are called as a policy enforcement check before/after each trap is forwarded.
00026  *
00027  * To minimize code duplication, a set of abstract wrapper classes is
00028  * provided, from which other wrappers may inherit. These abstract classes are
00029  * organized in the following hierarchy:
00030  *
00031  * BaseProxyHandler Wrapper
00032  * |                    | |
00033  * IndirectProxyHandler | |
00034  * |                  | | |
00035  * |      IndirectWrapper |
00036  * |                      |
00037  * DirectProxyHandler     |
00038  *                  |     |
00039  *            DirectWrapper
00040  */

00051     };

00077     /*
00078      * The function Wrapper::New takes a pointer to a Wrapper as the handler
00079      * object. It then passes it on to the function NewProxyObject, which
00080      * expects a pointer to a BaseProxyHandler as the handler object. We don't
00081      * want to change Wrapper::New to take a pointer to a BaseProxyHandler,
00082      * because that would allow the creation of wrappers with non-wrapper
00083      * handlers. Unfortunately, we can't inherit Wrapper from BaseProxyHandler,
00084      * since that would create a dreaded diamond, and we can't use dynamic_cast
00085      * to cast Wrapper to BaseProxyHandler, since that would require us to
00086      * compile with run time type information. Hence the need for this virtual
00087      * function.
00088      */
00089     virtual BaseProxyHandler *toBaseProxyHandler() = 0;
00090 
00091     /* Policy enforcement traps.
00092      *
00093      * enter() allows the policy to specify whether the caller may perform |act|
00094      * on the underlying object's |id| property. In the case when |act| is CALL,
00095      * |id| is generally JSID_VOID.
00096      *
00097      * The |act| parameter to enter() specifies the action being performed. GET,
00098      * SET, and CALL are self-explanatory, but PUNCTURE requires more
00099      * explanation:
00100      *
00101      * GET and SET allow for a very fine-grained security membrane, through
00102      * which access can be granted or denied on a per-property, per-object, and
00103      * per-action basis. Sometimes though, we just want to asks if we can access
00104      * _everything_ behind the wrapper barrier. For example, when the structured
00105      * clone algorithm runs up against a cross-compartment wrapper, it needs to
00106      * know whether it can enter the compartment and keep cloning, or whether it
00107      * should throw. This is the role of PUNCTURE.
00108      *
00109      * PUNCTURE allows the policy to specify whether the wrapper barrier may
00110      * be lifted - that is to say, whether the caller is allowed to access
00111      * anything that the wrapped object could access. This is a very powerful
00112      * permission, and thus should generally be denied for security wrappers
00113      * except under very special circumstances. When |act| is PUNCTURE, |id|
00114      * should be JSID_VOID.
00115      */
00116     virtual bool enter(JSContext *cx, JSObject *wrapper, jsid id, Action act,
00117                        bool *bp);
00118 };

00267 /*
00268  * Base class for security wrappers. A security wrapper is potentially hiding
00269  * all or part of some wrapped object thus SecurityWrapper defaults to denying
00270  * access to the wrappee. This is the opposite of Wrapper which tries to be
00271  * completely transparent.
00272  *
00273  * NB: Currently, only a few ProxyHandler operations are overridden to deny
00274  * access, relying on derived SecurityWrapper to block access when necessary.
00275  */

00277 class JS_FRIEND_API(SecurityWrapper) : public Base
00278 {
00279   public:
00280     SecurityWrapper(unsigned flags);

00283                             CallArgs args) MOZ_OVERRIDE;

00285     virtual bool regexp_toShared(JSContext *cx, JSObject *proxy, RegExpGuard *g) MOZ_OVERRIDE;
00286 };
