00308 /*
00309  * This hook captures high level script execution and function calls (JS or
00310  * native).  It is used by JS_SetExecuteHook to hook top level scripts and by
00311  * JS_SetCallHook to hook function calls.  It will get called twice per script
00312  * or function call: just before execution begins and just after it finishes.
00313  * In both cases the 'current' frame is that of the executing code.
00314  *
00315  * The 'before' param is JS_TRUE for the hook invocation before the execution
00316  * and JS_FALSE for the invocation after the code has run.
00317  *
00318  * The 'ok' param is significant only on the post execution invocation to
00319  * signify whether or not the code completed 'normally'.
00320  *
00321  * The 'closure' param is as passed to JS_SetExecuteHook or JS_SetCallHook
00322  * for the 'before'invocation, but is whatever value is returned from that
00323  * invocation for the 'after' invocation. Thus, the hook implementor *could*
00324  * allocate a structure in the 'before' invocation and return a pointer to that
00325  * structure. The pointer would then be handed to the hook for the 'after'
00326  * invocation. Alternately, the 'before' could just return the same value as
00327  * in 'closure' to cause the 'after' invocation to be called with the same
00328  * 'closure' value as the 'before'.
00329  *
00330  * Returning NULL in the 'before' hook will cause the 'after' hook *not* to
00331  * be called.
00332  */
00333 typedef void *
00334 (* JSInterpreterHook)(JSContext *cx, JSStackFrame *fp, JSBool before,
00335                       JSBool *ok, void *closure);
