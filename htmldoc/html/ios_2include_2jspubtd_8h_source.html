00023 /*
00024  * In release builds, jsid is defined to be an integral type. This
00025  * prevents many bugs from being caught at compile time. E.g.:
00026  *
00027  *  jsid id = ...
00028  *  if (id == JS_TRUE)  // error
00029  *    ...
00030  *
00031  *  size_t n = id;      // error
00032  *
00033  * To catch more errors, jsid is given a struct type in C++ debug builds.
00034  * Struct assignment and (in C++) operator== allow correct code to be mostly
00035  * oblivious to the change. This feature can be explicitly disabled in debug
00036  * builds by defining JS_NO_JSVAL_JSID_STRUCT_TYPES.
00037  */
00038 #ifdef __cplusplus
00039 
00040 # if defined(DEBUG) && !defined(JS_NO_JSVAL_JSID_STRUCT_TYPES)
00041 #  define JS_USE_JSID_STRUCT_TYPES
00042 # endif
00043 
00044 # ifdef JS_USE_JSID_STRUCT_TYPES
00045 struct jsid
00046 {
00047     size_t asBits;
00048     bool operator==(jsid rhs) const { return asBits == rhs.asBits; }
00049     bool operator!=(jsid rhs) const { return asBits != rhs.asBits; }
00050 };
00051 #  define JSID_BITS(id) (id.asBits)
00052 # else  /* defined(JS_USE_JSID_STRUCT_TYPES) */
00053 typedef ptrdiff_t jsid;
00054 #  define JSID_BITS(id) (id)
00055 # endif  /* defined(JS_USE_JSID_STRUCT_TYPES) */
00056 #else  /* defined(__cplusplus) */
00057 typedef ptrdiff_t jsid;
00058 # define JSID_BITS(id) (id)
00059 #endif
