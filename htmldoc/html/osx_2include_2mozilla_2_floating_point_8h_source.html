00015 /*
00016  * It's reasonable to ask why we have this header at all.  Don't isnan,
00017  * copysign, the built-in comparison operators, and the like solve these
00018  * problems?  Unfortunately, they don't.  We've found that various compilers
00019  * (MSVC, MSVC when compiling with PGO, and GCC on OS X, at least) miscompile
00020  * the standard methods in various situations, so we can't use them.  Some of
00021  * these compilers even have problems compiling seemingly reasonable bitwise
00022  * algorithms!  But with some care we've found algorithms that seem to not
00023  * trigger those compiler bugs.
00024  *
00025  * For the aforementioned reasons, be very wary of making changes to any of
00026  * these algorithms.  If you must make changes, keep a careful eye out for
00027  * compiler bustage, particularly PGO-specific bustage.
00028  *
00029  * Some users require that this file be C-compatible.  Unfortunately, this means
00030  * no mozilla namespace to contain everything, no detail namespace clarifying
00031  * MozDoublePun to be an internal data structure, and so on.
00032  */
00033 
00034 /*
00035  * These implementations all assume |double| is a 64-bit double format number
00036  * type, compatible with the IEEE-754 standard.  C/C++ don't require this to be
00037  * the case.  But we required this in implementations of these algorithms that
00038  * preceded this header, so we shouldn't break anything if we continue doing so.
00039  */
00040 MOZ_STATIC_ASSERT(sizeof(double) == sizeof(uint64_t), "double must be 64 bits");
00041 
00042 /*
00043  * Constant expressions in C can't refer to consts, unfortunately, so #define
00044  * these rather than use |const uint64_t|.
00045  */
00046 #define MOZ_DOUBLE_SIGN_BIT          0x8000000000000000ULL
00047 #define MOZ_DOUBLE_EXPONENT_BITS     0x7ff0000000000000ULL
00048 #define MOZ_DOUBLE_SIGNIFICAND_BITS  0x000fffffffffffffULL
00049 
00050 #define MOZ_DOUBLE_EXPONENT_BIAS   1023
00051 #define MOZ_DOUBLE_EXPONENT_SHIFT  52
00052 
00053 MOZ_STATIC_ASSERT((MOZ_DOUBLE_SIGN_BIT & MOZ_DOUBLE_EXPONENT_BITS) == 0,
00054                   "sign bit doesn't overlap exponent bits");
00055 MOZ_STATIC_ASSERT((MOZ_DOUBLE_SIGN_BIT & MOZ_DOUBLE_SIGNIFICAND_BITS) == 0,
00056                   "sign bit doesn't overlap significand bits");
00057 MOZ_STATIC_ASSERT((MOZ_DOUBLE_EXPONENT_BITS & MOZ_DOUBLE_SIGNIFICAND_BITS) == 0,
00058                   "exponent bits don't overlap significand bits");
00059 
00060 MOZ_STATIC_ASSERT((MOZ_DOUBLE_SIGN_BIT | MOZ_DOUBLE_EXPONENT_BITS | MOZ_DOUBLE_SIGNIFICAND_BITS)
00061                   == ~(uint64_t)0,
00062                   "all bits accounted for");

00068 /*
00069  * This union is NOT a public data structure, and it is not to be used outside
00070  * this file!
00071  */
00072 union MozDoublePun {
00073     /*
00074      * Every way to pun the bits of a double introduces an additional layer of
00075      * complexity, across a multitude of platforms, architectures, and ABIs.
00076      * Use *only* uint64_t to reduce complexity.  Don't add new punning here
00077      * without discussion!
00078      */
00079     uint64_t u;
00080     double d;
00081 };

00084 static MOZ_ALWAYS_INLINE int
00085 MOZ_DOUBLE_IS_NaN(double d)
00086 {
00087   union MozDoublePun pun;
00088   pun.d = d;
00089 
00090   /*
00091    * A double is NaN if all exponent bits are 1 and the significand contains at
00092    * least one non-zero bit.
00093    */
00094   return (pun.u & MOZ_DOUBLE_EXPONENT_BITS) == MOZ_DOUBLE_EXPONENT_BITS &&
00095          (pun.u & MOZ_DOUBLE_SIGNIFICAND_BITS) != 0;
00096 }

00099 static MOZ_ALWAYS_INLINE int
00100 MOZ_DOUBLE_IS_INFINITE(double d)
00101 {
00102   union MozDoublePun pun;
00103   pun.d = d;
00104 
00105   /* Infinities have all exponent bits set to 1 and an all-0 significand. */
00106   return (pun.u & ~MOZ_DOUBLE_SIGN_BIT) == MOZ_DOUBLE_EXPONENT_BITS;
00107 }

00110 static MOZ_ALWAYS_INLINE int
00111 MOZ_DOUBLE_IS_FINITE(double d)
00112 {
00113   union MozDoublePun pun;
00114   pun.d = d;
00115 
00116   /*
00117    * NaN and Infinities are the only non-finite doubles, and both have all
00118    * exponent bits set to 1.
00119    */
00120   return (pun.u & MOZ_DOUBLE_EXPONENT_BITS) != MOZ_DOUBLE_EXPONENT_BITS;
00121 }

00127 static MOZ_ALWAYS_INLINE int
00128 MOZ_DOUBLE_IS_NEGATIVE(double d)
00129 {
00130   union MozDoublePun pun;
00131   pun.d = d;
00132 
00133   MOZ_ASSERT(!MOZ_DOUBLE_IS_NaN(d), "NaN does not have a sign");
00134 
00135   /* The sign bit is set if the double is negative. */
00136   return (pun.u & MOZ_DOUBLE_SIGN_BIT) != 0;

00140 static MOZ_ALWAYS_INLINE int
00141 MOZ_DOUBLE_IS_NEGATIVE_ZERO(double d)
00142 {
00143   union MozDoublePun pun;
00144   pun.d = d;
00145 
00146   /* Only the sign bit is set if the double is -0. */

00151 static MOZ_ALWAYS_INLINE int_fast16_t
00152 MOZ_DOUBLE_EXPONENT(double d)
00153 {
00154   union MozDoublePun pun;
00155   pun.d = d;
00156 
00157   /*
00158    * The exponent component of a double is an unsigned number, biased from its
00159    * actual value.  Subtract the bias to retrieve the actual exponent.
00160    */
00161   return (int_fast16_t)((pun.u & MOZ_DOUBLE_EXPONENT_BITS) >> MOZ_DOUBLE_EXPONENT_SHIFT) -
00162                         MOZ_DOUBLE_EXPONENT_BIAS;
00163 }

00194 static MOZ_ALWAYS_INLINE double
00195 MOZ_DOUBLE_SPECIFIC_NaN(int signbit, uint64_t significand)
00196 {
00197   union MozDoublePun pun;
00198 
00199   MOZ_ASSERT(signbit == 0 || signbit == 1);
00200   MOZ_ASSERT((significand & ~MOZ_DOUBLE_SIGNIFICAND_BITS) == 0);
00201   MOZ_ASSERT(significand & MOZ_DOUBLE_SIGNIFICAND_BITS);
00202 
00203   pun.u = (signbit ? MOZ_DOUBLE_SIGN_BIT : 0) |
00204           MOZ_DOUBLE_EXPONENT_BITS |
00205           significand;
00206   MOZ_ASSERT(MOZ_DOUBLE_IS_NaN(pun.d));
00207   return pun.d;
00208 }
