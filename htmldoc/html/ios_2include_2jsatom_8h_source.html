00037 /*
00038  * Must not be used on atoms that are representable as integer jsids.
00039  * Prefer NameToId or AtomToId over this function:
00040  *
00041  * A PropertyName is an atom that does not contain an integer in the range
00042  * [0, UINT32_MAX]. However, jsid can only hold an integer in the range
00043  * [0, JSID_INT_MAX] (where JSID_INT_MAX == 2^31-1).  Thus, for the range of
00044  * integers (JSID_INT_MAX, UINT32_MAX], to represent as a jsid 'id', it must be
00045  * the case JSID_IS_ATOM(id) and !JSID_TO_ATOM(id)->isPropertyName().  In most
00046  * cases when creating a jsid, code does not have to care about this corner
00047  * case because:
00048  *
00049  * - When given an arbitrary JSAtom*, AtomToId must be used, which checks for
00050  *   integer atoms representable as integer jsids, and does this conversion.
00051  *
00052  * - When given a PropertyName*, NameToId can be used which which does not need
00053  *   to do any dynamic checks.
00054  *
00055  * Thus, it is only the rare third case which needs this function, which
00056  * handles any JSAtom* that is known not to be representable with an int jsid.
00057  */
00058 static JS_ALWAYS_INLINE jsid
00059 NON_INTEGER_ATOM_TO_JSID(JSAtom *atom)
00060 {
00061     JS_ASSERT(((size_t)atom & 0x7) == 0);
00062     jsid id = JSID_FROM_BITS((size_t)atom);
00063     JS_ASSERT(id == INTERNED_STRING_TO_JSID(NULL, (JSString*)atom));
00064     return id;
00065 }

00197 /*
00198  * On encodings:
00199  *
00200  * - Some string functions have an optional FlationCoding argument that allow
00201  *   the caller to force CESU-8 encoding handling.
00202  * - Functions that don't take a FlationCoding base their NormalEncoding
00203  *   behavior on the js_CStringsAreUTF8 value. NormalEncoding is either raw
00204  *   (simple zero-extension) or UTF-8 depending on js_CStringsAreUTF8.
00205  * - Functions that explicitly state their encoding do not use the
00206  *   js_CStringsAreUTF8 value.
00207  *
00208  * CESU-8 (Compatibility Encoding Scheme for UTF-16: 8-bit) is a variant of
00209  * UTF-8 that allows us to store any wide character string as a narrow
00210  * character string. For strings containing mostly ascii, it saves space.
