00024 /*
00025  * This class implements a counting Bloom filter as described at
00026  * <http://en.wikipedia.org/wiki/Bloom_filter#Counting_filters>, with
00027  * 8-bit counters.  This allows quick probabilistic answers to the
00028  * question "is object X in set Y?" where the contents of Y might not
00029  * be time-invariant.  The probabilistic nature of the test means that
00030  * sometimes the answer will be "yes" when it should be "no".  If the
00031  * answer is "no", then X is guaranteed not to be in Y.
00032  *
00033  * The filter is parametrized on KeySize, which is the size of the key
00034  * generated by each of hash functions used by the filter, in bits,
00035  * and the type of object T being added and removed.  T must implement
00036  * a |uint32_t hash() const| method which returns a uint32_t hash key
00037  * that will be used to generate the two separate hash functions for
00038  * the Bloom filter.  This hash key MUST be well-distributed for good
00039  * results!  KeySize is not allowed to be larger than 16.
00040  *
00041  * The filter uses exactly 2**KeySize bytes of memory.  From now on we
00042  * will refer to the memory used by the filter as M.
00043  *
00044  * The expected rate of incorrect "yes" answers depends on M and on
00045  * the number N of objects in set Y.  As long as N is small compared
00046  * to M, the rate of such answers is expected to be approximately
00047  * 4*(N/M)**2 for this filter.  In practice, if Y has a few hundred
00048  * elements then using a KeySize of 12 gives a reasonably low
00049  * incorrect answer rate.  A KeySize of 12 has the additional benefit
00050  * of using exactly one page for the filter in typical hardware
00051  * configurations.
00052  */
00053 
00054 template<unsigned KeySize, class T>
00055 class BloomFilter
00056 {
00057     /*
00058      * A counting Bloom filter with 8-bit counters.  For now we assume
00059      * that having two hash functions is enough, but we may revisit that
00060      * decision later.
00061      *
00062      * The filter uses an array with 2**KeySize entries.
00063      *
00064      * Assuming a well-distributed hash function, a Bloom filter with
00065      * array size M containing N elements and
00066      * using k hash function has expected false positive rate exactly
00067      *
00068      * $  (1 - (1 - 1/M)^{kN})^k  $
00069      *
00070      * because each array slot has a
00071      *
00072      * $  (1 - 1/M)^{kN}  $
00073      *
00074      * chance of being 0, and the expected false positive rate is the
00075      * probability that all of the k hash functions will hit a nonzero
00076      * slot.
00077      *
00078      * For reasonable assumptions (M large, kN large, which should both
00079      * hold if we're worried about false positives) about M and kN this
00080      * becomes approximately
00081      *
00082      * $$  (1 - \exp(-kN/M))^k   $$
00083      *
00084      * For our special case of k == 2, that's $(1 - \exp(-2N/M))^2$,
00085      * or in other words
00086      *
00087      * $$    N/M = -0.5 * \ln(1 - \sqrt(r))   $$
00088      *
00089      * where r is the false positive rate.  This can be used to compute
00090      * the desired KeySize for a given load N and false positive rate r.
00091      *
00092      * If N/M is assumed small, then the false positive rate can
00093      * further be approximated as 4*N^2/M^2.  So increasing KeySize by
00094      * 1, which doubles M, reduces the false positive rate by about a
00095      * factor of 4, and a false positive rate of 1% corresponds to
00096      * about M/N == 20.
00097      *
00098      * What this means in practice is that for a few hundred keys using a
00099      * KeySize of 12 gives false positive rates on the order of 0.25-4%.
00100      *
00101      * Similarly, using a KeySize of 10 would lead to a 4% false
00102      * positive rate for N == 100 and to quite bad false positive
00103      * rates for larger N.
00104      */
00105   public:
00106     BloomFilter() {
00107         MOZ_STATIC_ASSERT(KeySize <= keyShift, "KeySize too big");
00108 
00109         // Should we have a custom operator new using calloc instead and
00110         // require that we're allocated via the operator?
00111         clear();
00112     }
00113 
00114     /*
00115      * Clear the filter.  This should be done before reusing it, because
00116      * just removing all items doesn't clear counters that hit the upper
00117      * bound.
00118      */
00119     void clear();
00120 
00121     /*
00122      * Add an item to the filter.
00123      */
00124     void add(const T* t);
00125 
00126     /*
00127      * Remove an item from the filter.
00128      */
00129     void remove(const T* t);
00130 
00131     /*
00132      * Check whether the filter might contain an item.  This can
00133      * sometimes return true even if the item is not in the filter,
00134      * but will never return false for items that are actually in the
00135      * filter.
00136      */
00137     bool mightContain(const T* t) const;
00138 
00139     /*
00140      * Methods for add/remove/contain when we already have a hash computed
00141      */
00142     void add(uint32_t hash);
00143     void remove(uint32_t hash);
00144     bool mightContain(uint32_t hash) const;
00145 
00146   private:
00147     static const size_t arraySize = (1 << KeySize);
00148     static const uint32_t keyMask = (1 << KeySize) - 1;
00149     static const uint32_t keyShift = 16;
00150 
00151     static uint32_t hash1(uint32_t hash) { return hash & keyMask; }
00152     static uint32_t hash2(uint32_t hash) { return (hash >> keyShift) & keyMask; }
00153 
00154     uint8_t& firstSlot(uint32_t hash) { return counters[hash1(hash)]; }
00155     uint8_t& secondSlot(uint32_t hash) { return counters[hash2(hash)]; }
00156     const uint8_t& firstSlot(uint32_t hash) const { return counters[hash1(hash)]; }
00157     const uint8_t& secondSlot(uint32_t hash) const { return counters[hash2(hash)]; }
00158 
00159     static bool full(const uint8_t& slot) { return slot == UINT8_MAX; }
00160 
00161     uint8_t counters[arraySize];
00162 };

00196   // If the slots are full, we don't know whether we bumped them to be
