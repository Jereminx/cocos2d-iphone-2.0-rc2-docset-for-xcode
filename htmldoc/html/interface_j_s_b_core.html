        jsval rval;
        JSString *str;
        JSBool ok;
        const char *filename = "noname";
        uint32_t lineno = 0;
        if (outVal == NULL) {
                outVal = &rval;
        }
        const char *cstr = [string UTF8String];
        ok = JS_EvaluateScript( _cx, _object, cstr, (unsigned)strlen(cstr), filename, lineno, outVal);

                CCLOGWARN(@"error evaluating script:%@", string);

        return ok;
