00030 /*
00031  * We partition the ways to refer to a property into three: by an index
00032  * (uint32_t); by a string whose characters do not represent an index
00033  * (PropertyName, see vm/String.h); and by various special values.
00034  *
00035  * Special values are encoded using SpecialId, which is layout-compatible but
00036  * non-interconvertible with jsid.  A SpecialId may be: an object (used by E4X
00037  * and perhaps eventually by Harmony-proposed private names); JSID_VOID, which
00038  * does not occur in JS scripts but may be used to indicate the absence of a
00039  * valid identifier; or JS_DEFAULT_XML_NAMESPACE_ID, if E4X is enabled.
00040  */
00041 class SpecialId
00042 {
00043     uintptr_t bits;
00044 
00045     /* Needs access to raw bits. */
00046     friend JS_ALWAYS_INLINE jsid SPECIALID_TO_JSID(const SpecialId &sid);
00047     friend class PropertyId;
00048 
00049     static const uintptr_t TYPE_VOID = JSID_TYPE_VOID;
00050     static const uintptr_t TYPE_OBJECT = JSID_TYPE_OBJECT;
00051     static const uintptr_t TYPE_DEFAULT_XML_NAMESPACE = JSID_TYPE_DEFAULT_XML_NAMESPACE;
00052     static const uintptr_t TYPE_MASK = JSID_TYPE_MASK;
00053 
00054     SpecialId(uintptr_t bits) : bits(bits) { }
00055 
00056   public:
00057     SpecialId() : bits(TYPE_VOID) { }
00058 
00059     /* Object-valued */
00060 
00061     SpecialId(JSObject &obj)
00062       : bits(uintptr_t(&obj) | TYPE_OBJECT)
00063     {
00064         JS_ASSERT(&obj != NULL);
00065         JS_ASSERT((uintptr_t(&obj) & TYPE_MASK) == 0);
00066     }
00067 
00068     bool isObject() const {
00069         return (bits & TYPE_MASK) == TYPE_OBJECT && bits != TYPE_OBJECT;
00070     }
00071 
00072     JSObject *toObject() const {
00073         JS_ASSERT(isObject());
00074         return reinterpret_cast<JSObject *>(bits & ~TYPE_MASK);
00075     }
00076 
00077     /* Empty */
00078 
00079     static SpecialId empty() {
00080         SpecialId sid(TYPE_OBJECT);
00081         JS_ASSERT(sid.isEmpty());
00082         return sid;
00083     }
00084 
00085     bool isEmpty() const {
00086         return bits == TYPE_OBJECT;
00087     }
00088 
00089     /* Void */
00090 
00091     static SpecialId voidId() {
00092         SpecialId sid(TYPE_VOID);
00093         JS_ASSERT(sid.isVoid());
00094         return sid;
00095     }
00096 
00097     bool isVoid() const {
00098         return bits == TYPE_VOID;
00099     }
00100 
00101     /* Default XML namespace */
00102 
00103     static SpecialId defaultXMLNamespace() {
00104         SpecialId sid(TYPE_DEFAULT_XML_NAMESPACE);
00105         JS_ASSERT(sid.isDefaultXMLNamespace());
00106         return sid;
00107     }
00108 
00109     bool isDefaultXMLNamespace() const {
00110         return bits == TYPE_DEFAULT_XML_NAMESPACE;
00111     }
00112 };

00114 static JS_ALWAYS_INLINE jsid
00115 SPECIALID_TO_JSID(const SpecialId &sid)
00116 {
00117     jsid id;
00118     JSID_BITS(id) = sid.bits;
00119     JS_ASSERT_IF(sid.isObject(), JSID_IS_OBJECT(id) && JSID_TO_OBJECT(id) == sid.toObject());
00120     JS_ASSERT_IF(sid.isVoid(), JSID_IS_VOID(id));
00121     JS_ASSERT_IF(sid.isEmpty(), JSID_IS_EMPTY(id));
00122     JS_ASSERT_IF(sid.isDefaultXMLNamespace(), JSID_IS_DEFAULT_XML_NAMESPACE(id));
00123     return id;
00124 }

00126 static JS_ALWAYS_INLINE bool
00127 JSID_IS_SPECIAL(jsid id)
00128 {
00129     return JSID_IS_OBJECT(id) || JSID_IS_EMPTY(id) || JSID_IS_VOID(id) ||
00130            JSID_IS_DEFAULT_XML_NAMESPACE(id);

00133 static JS_ALWAYS_INLINE SpecialId
00134 JSID_TO_SPECIALID(jsid id)
00135 {
00136     JS_ASSERT(JSID_IS_SPECIAL(id));
00137     if (JSID_IS_OBJECT(id))
00138         return SpecialId(*JSID_TO_OBJECT(id));
00139     if (JSID_IS_EMPTY(id))
00140         return SpecialId::empty();
00141     if (JSID_IS_VOID(id))
00142         return SpecialId::voidId();
00143     JS_ASSERT(JSID_IS_DEFAULT_XML_NAMESPACE(id));
00144     return SpecialId::defaultXMLNamespace();
00145 }
