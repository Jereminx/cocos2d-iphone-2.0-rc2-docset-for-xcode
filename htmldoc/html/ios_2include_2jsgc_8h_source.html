00183 /*
00184  * ArenaList::head points to the start of the list. Normally cursor points
00185  * to the first arena in the list with some free things and all arenas
00186  * before cursor are fully allocated. However, as the arena currently being
00187  * allocated from is considered full while its list of free spans is moved
00188  * into the freeList, during the GC or cell enumeration, when an
00189  * unallocated freeList is moved back to the arena, we can see an arena
00190  * with some free cells before the cursor. The cursor is an indirect
00191  * pointer to allow for efficient list insertion at the cursor point and
00192  * other list manipulations.
00193  */

00196     ArenaHeader     **cursor;
00197 
00198     ArenaList() {
00199         clear();
00200     }
00201 
00202     void clear() {
00203         head = NULL;
00204         cursor = &head;
00205     }

00207     void insert(ArenaHeader *arena);
00208 };

00210 struct ArenaLists {
00211 
00212   private:
00213     /*
00214      * For each arena kind its free list is represented as the first span with
00215      * free things. Initially all the spans are initialized as empty. After we
00216      * find a new arena with available things we move its first free span into
00217      * the list and set the arena as fully allocated. way we do not need to
00218      * update the arena header after the initial allocation. When starting the
00219      * GC we only move the head of the of the list of spans back to the arena
00220      * only for the arena that was not fully allocated.
00221      */
00222     FreeSpan       freeLists[FINALIZE_LIMIT];
00223 
00224     ArenaList      arenaLists[FINALIZE_LIMIT];
00225 
00226     /*
00227      * The background finalization adds the finalized arenas to the list at
00228      * the *cursor position. backgroundFinalizeState controls the interaction
00229      * between the GC lock and the access to the list from the allocation
00230      * thread.
00231      *
00232      * BFS_DONE indicates that the finalizations is not running or cannot
00233      * affect this arena list. The allocation thread can access the list
00234      * outside the GC lock.
00235      *
00236      * In BFS_RUN and BFS_JUST_FINISHED the allocation thread must take the
00237      * lock. The former indicates that the finalization still runs. The latter
00238      * signals that finalization just added to the list finalized arenas. In
00239      * that case the lock effectively serves as a read barrier to ensure that
00240      * the allocation thread see all the writes done during finalization.
00241      */
00242     enum BackgroundFinalizeState {
00243         BFS_DONE,
00244         BFS_RUN,
00245         BFS_JUST_FINISHED
00246     };
00247 
00248     volatile uintptr_t backgroundFinalizeState[FINALIZE_LIMIT];
00249 
00250   public:
00251     /* For each arena kind, a list of arenas remaining to be swept. */
00252     ArenaHeader *arenaListsToSweep[FINALIZE_LIMIT];
00253 
00254   public:
00255     ArenaLists() {
00256         for (size_t i = 0; i != FINALIZE_LIMIT; ++i)
00257             freeLists[i].initAsEmpty();
00258         for (size_t i = 0; i != FINALIZE_LIMIT; ++i)
00259             backgroundFinalizeState[i] = BFS_DONE;
00260         for (size_t i = 0; i != FINALIZE_LIMIT; ++i)
00261             arenaListsToSweep[i] = NULL;
00262     }
00263 
00264     ~ArenaLists() {
00265         for (size_t i = 0; i != FINALIZE_LIMIT; ++i) {
00266             /*
00267              * We can only call this during the shutdown after the last GC when
00268              * the background finalization is disabled.
00269              */
00270             JS_ASSERT(backgroundFinalizeState[i] == BFS_DONE);
00271             ArenaHeader **headp = &arenaLists[i].head;
00272             while (ArenaHeader *aheader = *headp) {
00273                 *headp = aheader->next;
00274                 aheader->chunk()->releaseArena(aheader);
00275             }
00276         }
00277     }

00292         for (size_t i = 0; i != FINALIZE_LIMIT; ++i) {
00293             /*
00294              * The arena cannot be empty if the background finalization is not yet

00759     bool init(size_t ballastcap) {
00760         JS_ASSERT(!stack);
00761 
00762         if (ballastcap == 0)
00763             return true;
00764 
00765         ballast = (T *)js_malloc(sizeof(T) * ballastcap);
00766         if (!ballast)
00767             return false;
00768         ballastLimit = ballast + ballastcap;
00769         initFromBallast();

00773     void initFromBallast() {
00774         stack = ballast;
00775         limit = ballastLimit;
00776         if (size_t(limit - stack) > sizeLimit)
00777             limit = stack + sizeLimit;
00778         tos = stack;
00779     }

00782         JS_ASSERT(isEmpty());

00785         reset();
00786     }

00788     bool push(T item) {
00789         if (tos == limit) {
00790             if (!enlarge())
00791                 return false;
00792         }
00793         JS_ASSERT(tos < limit);
00794         *tos++ = item;
00795         return true;
00796     }
00797 
00798     bool push(T item1, T item2, T item3) {
00799         T *nextTos = tos + 3;
00800         if (nextTos > limit) {
00801             if (!enlarge())
00802                 return false;
00803             nextTos = tos + 3;
00804         }
00805         JS_ASSERT(nextTos <= limit);
00806         tos[0] = item1;
00807         tos[1] = item2;
00808         tos[2] = item3;
00809         tos = nextTos;
00810         return true;
00811     }

00813     bool isEmpty() const {
00814         return tos == stack;
00815     }
00816 
00817     T pop() {
00818         JS_ASSERT(!isEmpty());

00820     }

00822     ptrdiff_t position() const {
00823         return tos - stack;
00824     }

00826     void reset() {
00827         if (stack != ballast)
00828             js_free(stack);
00829         initFromBallast();
00830         JS_ASSERT(stack == ballast);
00831     }
00832 
00833     bool enlarge() {
00834         size_t tosIndex = tos - stack;
00835         size_t cap = limit - stack;
00836         if (cap == sizeLimit)
00837             return false;
00838         size_t newcap = cap * 2;
00839         if (newcap == 0)
00840             newcap = 32;
00841         if (newcap > sizeLimit)
00842             newcap = sizeLimit;
00843 
00844         T *newStack;
00845         if (stack == ballast) {
00846             newStack = (T *)js_malloc(sizeof(T) * newcap);
00847             if (!newStack)
00848                 return false;
00849             for (T *src = stack, *dst = newStack; src < tos; )
00850                 *dst++ = *src++;
00851         } else {
00852             newStack = (T *)js_realloc(stack, sizeof(T) * newcap);
00853             if (!newStack)
00854                 return false;
00855         }
00856         stack = newStack;
00857         tos = stack + tosIndex;
00858         limit = newStack + newcap;
00859         return true;
00860     }
