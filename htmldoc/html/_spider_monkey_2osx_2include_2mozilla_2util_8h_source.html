00164 /*
00165  * This utility pales in comparison to Boost's aligned_storage. The utility
00166  * simply assumes that uint64_t is enough alignment for anyone. This may need
00167  * to be extended one day...
00168  *
00169  * As an important side effect, pulling the storage into this template is
00170  * enough obfuscation to confuse gcc's strict-aliasing analysis into not giving
00171  * false negatives when we cast from the char buffer to whatever type we've
00172  * constructed using the bytes.
00173  */
00174 template<size_t nbytes>
00175 struct AlignedStorage
00176 {
00177     union U {
00178       char bytes[nbytes];
00179       uint64_t _;
00180     } u;
00181 
00182     const void* addr() const { return u.bytes; }
00183     void* addr() { return u.bytes; }
00184 };
00185 
00186 template<class T>
00187 struct AlignedStorage2
00188 {
00189     union U {
00190       char bytes[sizeof(T)];
00191       uint64_t _;
00192     } u;
00193 
00194     const T* addr() const { return reinterpret_cast<const T*>(u.bytes); }
00195     T* addr() { return static_cast<T*>(static_cast<void*>(u.bytes)); }
00196 };

00198 /*
00199  * Small utility for lazily constructing objects without using dynamic storage.
00200  * When a Maybe<T> is constructed, it is |empty()|, i.e., no value of T has
00201  * been constructed and no T destructor will be called when the Maybe<T> is
00202  * destroyed. Upon calling |construct|, a T object will be constructed with the
00203  * given arguments and that object will be destroyed when the owning Maybe<T>
00204  * is destroyed.
00205  *
00206  * N.B. GCC seems to miss some optimizations with Maybe and may generate extra
00207  * branches/loads/stores. Use with caution on hot paths.
00208  */
00209 template<class T>
00210 class Maybe
00211 {
00212     AlignedStorage2<T> storage;
00213     bool constructed;
00214 
00215     T& asT() { return *storage.addr(); }
00216 
00217   public:
00218     Maybe() { constructed = false; }
00219     ~Maybe() { if (constructed) asT().~T(); }
00220 
00221     bool empty() const { return !constructed; }
00222 
00223     void construct() {
00224       MOZ_ASSERT(!constructed);
00225       new (storage.addr()) T();
00226       constructed = true;
00227     }
00228 
00229     template<class T1>
00230     void construct(const T1& t1) {
00231       MOZ_ASSERT(!constructed);
00232       new (storage.addr()) T(t1);
00233       constructed = true;
00234     }
00235 
00236     template<class T1, class T2>
00237     void construct(const T1& t1, const T2& t2) {
00238       MOZ_ASSERT(!constructed);
00239       new (storage.addr()) T(t1, t2);
00240       constructed = true;
00241     }
00242 
00243     template<class T1, class T2, class T3>
00244     void construct(const T1& t1, const T2& t2, const T3& t3) {
00245       MOZ_ASSERT(!constructed);
00246       new (storage.addr()) T(t1, t2, t3);
00247       constructed = true;
00248     }
00249 
00250     template<class T1, class T2, class T3, class T4>
00251     void construct(const T1& t1, const T2& t2, const T3& t3, const T4& t4) {
00252       MOZ_ASSERT(!constructed);
00253       new (storage.addr()) T(t1, t2, t3, t4);
00254       constructed = true;
00255     }
00256 
00257     T* addr() {
00258       MOZ_ASSERT(constructed);
00259       return &asT();
00260     }
00261 
00262     T& ref() {
00263       MOZ_ASSERT(constructed);
00264       return asT();
00265     }
00266 
00267     const T& ref() const {
00268       MOZ_ASSERT(constructed);
00269       return const_cast<Maybe*>(this)->asT();
00270     }
00271 
00272     void destroy() {
00273       ref().~T();

00275     }

00287 /*
00288  * Safely subtract two pointers when it is known that end >= begin.  This avoids
00289  * the common compiler bug that if (size_t(end) - size_t(begin)) has the MSB
00290  * set, the unsigned subtraction followed by right shift will produce -1, or
00291  * size_t(-1), instead of the real difference.
00292  */
00293 template<class T>
00294 MOZ_ALWAYS_INLINE size_t
00295 PointerRangeSize(T* begin, T* end)
00296 {
00297   MOZ_ASSERT(end >= begin);
00298   return (size_t(end) - size_t(begin)) / sizeof(T);
00299 }
