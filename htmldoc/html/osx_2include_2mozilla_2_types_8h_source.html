00029 /*
00030  * MOZ_EXPORT_API is used to declare and define a method which is externally
00031  * visible to users of the current library.  It encapsulates various decorations
00032  * needed to properly export the method's symbol.  MOZ_EXPORT_DATA serves the
00033  * same purpose for data.
00034  *
00035  *   api.h:
00036  *     extern MOZ_EXPORT_API(int) MeaningOfLife(void);
00037  *     extern MOZ_EXPORT_DATA(int) LuggageCombination;
00038  *
00039  *   api.c:
00040  *     MOZ_EXPORT_API(int) MeaningOfLife(void) { return 42; }
00041  *     MOZ_EXPORT_DATA(int) LuggageCombination = 12345;
00042  *
00043  * If you are merely sharing a method across files, just use plain |extern|.
00044  * These macros are designed for use by library interfaces -- not for normal
00045  * methods or data used cross-file.
00046  */
00047 #if defined(WIN32) || defined(XP_OS2)
00048 #  define MOZ_EXPORT_API(type)    __declspec(dllexport) type
00049 #  define MOZ_EXPORT_DATA(type)   __declspec(dllexport) type
00050 #else /* Unix */
00051 #  ifdef HAVE_VISIBILITY_ATTRIBUTE
00052 #    define MOZ_EXTERNAL_VIS       __attribute__((visibility("default")))
00053 #  elif defined(__SUNPRO_C) || defined(__SUNPRO_CC)
00054 #    define MOZ_EXTERNAL_VIS      __global
00055 #  else
00056 #    define MOZ_EXTERNAL_VIS
00057 #  endif
00058 #  define MOZ_EXPORT_API(type)    MOZ_EXTERNAL_VIS type
00059 #  define MOZ_EXPORT_DATA(type)   MOZ_EXTERNAL_VIS type
00060 #endif
00061 
00062 /*
00063  * Whereas implementers use MOZ_EXPORT_API and MOZ_EXPORT_DATA to declare and
00064  * define library symbols, users use MOZ_IMPORT_API and MOZ_IMPORT_DATA to
00065  * access them.  Most often the implementer of the library will expose an API
00066  * macro which expands to either the export or import version of the macro,
00067  * depending upon the compilation mode.
00068  */
00069 #ifdef _WIN32
00070 #  if defined(__MWERKS__)
00071 #    define MOZ_IMPORT_API(x)    x
00072 #  else
00073 #    define MOZ_IMPORT_API(x)    __declspec(dllimport) x
00074 #  endif
00075 #elif defined(XP_OS2)
00076 #  define MOZ_IMPORT_API(x)     __declspec(dllimport) x
00077 #else
00078 #  define MOZ_IMPORT_API(x)     MOZ_EXPORT_API(x)
00079 #endif
00080 
00081 #if defined(_WIN32) && !defined(__MWERKS__)
00082 #  define MOZ_IMPORT_DATA(x)     __declspec(dllimport) x
00083 #elif defined(XP_OS2)
00084 #  define MOZ_IMPORT_DATA(x)     __declspec(dllimport) x
00085 #else
00086 #  define MOZ_IMPORT_DATA(x)     MOZ_EXPORT_DATA(x)
00087 #endif

00111 /*
00112  * C symbols in C++ code must be declared immediately within |extern "C"|
00113  * blocks.  However, in C code, they need not be declared specially.  This
00114  * difference is abstracted behind the MOZ_BEGIN_EXTERN_C and MOZ_END_EXTERN_C
00115  * macros, so that the user need not know whether he is being used in C or C++
00116  * code.
00117  *
00118  *   MOZ_BEGIN_EXTERN_C
00119  *
00120  *   extern MOZ_EXPORT_API(int) MostRandomNumber(void);
00121  *   ...other declarations...
00122  *
00123  *   MOZ_END_EXTERN_C
00124  *
00125  * This said, it is preferable to just use |extern "C"| in C++ header files for
00126  * its greater clarity.
00127  */
00128 #ifdef __cplusplus
00129 #  define MOZ_BEGIN_EXTERN_C    extern "C" {
00130 #  define MOZ_END_EXTERN_C      }
