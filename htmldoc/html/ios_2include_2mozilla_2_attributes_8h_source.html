00173 /*
00174  * MOZ_DELETE, specified immediately prior to the ';' terminating an undefined-
00175  * method declaration, attempts to delete that method from the corresponding
00176  * class.  An attempt to use the method will always produce an error *at compile
00177  * time* (instead of sometimes as late as link time) when this macro can be
00178  * implemented.  For example, you can use MOZ_DELETE to produce classes with no
00179  * implicit copy constructor or assignment operator:
00180  *
00181  *   struct NonCopyable
00182  *   {
00183  *     private:
00184  *       NonCopyable(const NonCopyable& other) MOZ_DELETE;
00185  *       void operator=(const NonCopyable& other) MOZ_DELETE;
00186  *   };
00187  *
00188  * If MOZ_DELETE can't be implemented for the current compiler, use of the
00189  * annotated method will still cause an error, but the error might occur at link
00190  * time in some cases rather than at compile time.
00191  *
00192  * MOZ_DELETE relies on C++11 functionality not universally implemented.  As a
00193  * backstop, method declarations using MOZ_DELETE should be private.
00194  */
00195 #if defined(MOZ_HAVE_CXX11_DELETE)
00196 #  define MOZ_DELETE            = delete
00197 #else
00198 #  define MOZ_DELETE            /* no support */
00199 #endif

00201 /*
00202  * MOZ_OVERRIDE explicitly indicates that a virtual member function in a class
00203  * overrides a member function of a base class, rather than potentially being a
00204  * new member function.  MOZ_OVERRIDE should be placed immediately before the
00205  * ';' terminating the member function's declaration, or before '= 0;' if the
00206  * member function is pure.  If the member function is defined in the class
00207  * definition, it should appear before the opening brace of the function body.
00208  *
00209  *   class Base
00210  *   {
00211  *     public:
00212  *       virtual void f() = 0;
00213  *   };
00214  *   class Derived1 : public Base
00215  *   {
00216  *     public:
00217  *       virtual void f() MOZ_OVERRIDE;
00218  *   };
00219  *   class Derived2 : public Base
00220  *   {
00221  *     public:
00222  *       virtual void f() MOZ_OVERRIDE = 0;
00223  *   };
00224  *   class Derived3 : public Base
00225  *   {
00226  *     public:
00227  *       virtual void f() MOZ_OVERRIDE { }
00228  *   };
00229  *
00230  * In compilers supporting C++11 override controls, MOZ_OVERRIDE *requires* that
00231  * the function marked with it override a member function of a base class: it
00232  * is a compile error if it does not.  Otherwise MOZ_OVERRIDE does not affect
00233  * semantics and merely documents the override relationship to the reader (but
00234  * of course must still be used correctly to not break C++11 compilers).
00235  */
00236 #if defined(MOZ_HAVE_CXX11_OVERRIDE)
00237 #  define MOZ_OVERRIDE          override
00238 #else
00239 #  define MOZ_OVERRIDE          /* no support */
00240 #endif

00242 /*
00243  * MOZ_FINAL indicates that some functionality cannot be overridden through
00244  * inheritance.  It can be used to annotate either classes/structs or virtual
00245  * member functions.
00246  *
00247  * To annotate a class/struct with MOZ_FINAL, place MOZ_FINAL immediately after
00248  * the name of the class, before the list of classes from which it derives (if
00249  * any) and before its opening brace.  MOZ_FINAL must not be used to annotate
00250  * unnamed classes or structs.  (With some compilers, and with C++11 proper, the
00251  * underlying expansion is ambiguous with specifying a class name.)
00252  *
00253  *   class Base MOZ_FINAL
00254  *   {
00255  *     public:
00256  *       Base();
00257  *       ~Base();
00258  *       virtual void f() { }
00259  *   };
00260  *   // This will be an error in some compilers:
00261  *   class Derived : public Base
00262  *   {
00263  *     public:
00264  *       ~Derived() { }
00265  *   };
00266  *
00267  * One particularly common reason to specify MOZ_FINAL upon a class is to tell
00268  * the compiler that it's not dangerous for it to have a non-virtual destructor
00269  * yet have one or more virtual functions, silencing the warning it might emit
00270  * in this case.  Suppose Base above weren't annotated with MOZ_FINAL.  Because
00271  * ~Base() is non-virtual, an attempt to delete a Derived* through a Base*
00272  * wouldn't call ~Derived(), so any cleanup ~Derived() might do wouldn't happen.
00273  * (Formally C++ says behavior is undefined, but compilers will likely just call
00274  * ~Base() and not ~Derived().)  Specifying MOZ_FINAL tells the compiler that
00275  * it's safe for the destructor to be non-virtual.
00276  *
00277  * In compilers implementing final controls, it is an error to inherit from a
00278  * class annotated with MOZ_FINAL.  In other compilers it serves only as
00279  * documentation.
00280  *
00281  * To annotate a virtual member function with MOZ_FINAL, place MOZ_FINAL
00282  * immediately before the ';' terminating the member function's declaration, or
00283  * before '= 0;' if the member function is pure.  If the member function is
00284  * defined in the class definition, it should appear before the opening brace of
00285  * the function body.  (This placement is identical to that for MOZ_OVERRIDE.
00286  * If both are used, they should appear in the order 'MOZ_FINAL MOZ_OVERRIDE'
00287  * for consistency.)
00288  *
00289  *   class Base
00290  *   {
00291  *     public:
00292  *       virtual void f() MOZ_FINAL;
00293  *   };
00294  *   class Derived
00295  *   {
00296  *     public:
00297  *       // This will be an error in some compilers:
00298  *       virtual void f();
00299  *   };
00300  *
00301  * In compilers implementing final controls, it is an error for a derived class
00302  * to override a method annotated with MOZ_FINAL.  In other compilers it serves
00303  * only as documentation.
00304  */
00305 #if defined(MOZ_HAVE_CXX11_FINAL)
00306 #  define MOZ_FINAL             MOZ_HAVE_CXX11_FINAL
00307 #else
00308 #  define MOZ_FINAL             /* no support */
00309 #endif
