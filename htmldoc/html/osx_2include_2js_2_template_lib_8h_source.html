00079 /*
00080  * Produce an N-bit mask, where N <= BitSize<size_t>::result.  Handle the
00081  * language-undefined edge case when N = BitSize<size_t>::result.
00082  */
00083 template <size_t N> struct NBitMask {
00084     typedef typename StaticAssert<N < BitSize<size_t>::result>::result _;
00085     static const size_t result = (size_t(1) << N) - 1;
00086 };
00087 template <> struct NBitMask<BitSize<size_t>::result> {
00088     static const size_t result = size_t(-1);

00091 /*
00092  * For the unsigned integral type size_t, compute a mask M for N such that
00093  * for all X, !(X & M) implies X * N will not overflow (w.r.t size_t)
00094  */
00095 template <size_t N> struct MulOverflowMask {
00096     static const size_t result =
00097         ~NBitMask<BitSize<size_t>::result - CeilingLog2<N>::result>::result;
00098 };
00099 template <> struct MulOverflowMask<0> { /* Error */ };
00100 template <> struct MulOverflowMask<1> { static const size_t result = 0; };

00102 /*
00103  * Generate a mask for T such that if (X & sUnsafeRangeSizeMask), an X-sized
00104  * array of T's is big enough to cause a ptrdiff_t overflow when subtracting
00105  * a pointer to the end of the array from the beginning.

00107 template <class T> struct UnsafeRangeSizeMask {
