00186     // If you add a new number, remember to update the constructor!
00187 
00188     // Here's a useful breakdown of the GC heap.
00189     //
00190     // - rtStats.gcHeapChunkTotal
00191     //   - decommitted bytes
00192     //     - rtStats.gcHeapDecommittedArenas (decommitted arenas in non-empty chunks)
00193     //   - unused bytes
00194     //     - rtStats.gcHeapUnusedChunks (empty chunks)
00195     //     - rtStats.gcHeapUnusedArenas (empty arenas within non-empty chunks)
00196     //     - rtStats.total.gcHeapUnusedGcThings (empty GC thing slots within non-empty arenas)
00197     //   - used bytes
00198     //     - rtStats.gcHeapChunkAdmin
00199     //     - rtStats.total.gcHeapArenaAdmin
00200     //     - rtStats.gcHeapGcThings (in-use GC things)
00201     //
00202     // It's possible that some arenas in empty chunks may be decommitted, but
00203     // we don't count those under rtStats.gcHeapDecommittedArenas because (a)
00204     // it's rare, and (b) this means that rtStats.gcHeapUnusedChunks is a
00205     // multiple of the chunk size, which is good.
00206

00208     size_t gcHeapDecommittedArenas;
00209     size_t gcHeapUnusedChunks;
