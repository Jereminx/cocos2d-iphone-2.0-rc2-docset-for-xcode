00169 /* Special munging to support doing things the 'cygwin' way:
00170  * 'Normal' png-on-win32 defines/defaults:
00171  *   PNG_BUILD_DLL -- building dll
00172  *   PNG_USE_DLL   -- building an application, linking to dll
00173  *   (no define)   -- building static library, or building an
00174  *                    application and linking to the static lib
00175  * 'Cygwin' defines/defaults:
00176  *   PNG_BUILD_DLL -- (ignored) building the dll
00177  *   (no define)   -- (ignored) building an application, linking to the dll
00178  *   PNG_STATIC    -- (ignored) building the static lib, or building an
00179  *                    application that links to the static lib.
00180  *   ALL_STATIC    -- (ignored) building various static libs, or building an
00181  *                    application that links to the static libs.
00182  * Thus,
00183  * a cygwin user should define either PNG_BUILD_DLL or PNG_STATIC, and
00184  * this bit of #ifdefs will define the 'correct' config variables based on
00185  * that. If a cygwin user *wants* to define 'PNG_USE_DLL' that's okay, but
00186  * unnecessary.
00187  *
00188  * Also, the precedence order is:
00189  *   ALL_STATIC (since we can't #undef something outside our namespace)
00190  *   PNG_BUILD_DLL
00191  *   PNG_STATIC
00192  *   (nothing) == PNG_USE_DLL
00193  *
00194  * CYGWIN (2002-01-20): The preceding is now obsolete. With the advent
00195  *   of auto-import in binutils, we no longer need to worry about
00196  *   __declspec(dllexport) / __declspec(dllimport) and friends.  Therefore,
00197  *   we don't need to worry about PNG_STATIC or ALL_STATIC when it comes
00198  *   to __declspec() stuff.  However, we DO need to worry about
00199  *   PNG_BUILD_DLL and PNG_STATIC because those change some defaults
00200  *   such as CONSOLE_IO and whether GLOBAL_ARRAYS are allowed.
00201  */

00212 #    endif

00215 #    endif
