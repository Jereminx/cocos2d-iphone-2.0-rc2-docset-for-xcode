00250 /*
00251 ** Macro version of JS_CeilingLog2: Compute the log of the least power of
00252 ** 2 greater than or equal to _n. The result is returned in _log2.
00253 */
00254 #ifdef JS_HAS_BUILTIN_BITSCAN32
00255 /*
00256  * Use intrinsic function or count-leading-zeros to calculate ceil(log2(_n)).
00257  * The macro checks for "n <= 1" and not "n != 0" as js_bitscan_clz32(0) is
00258  * undefined.
00259  */
00260 # define JS_CEILING_LOG2(_log2,_n)                                            \
00261     JS_BEGIN_MACRO                                                            \
00262         unsigned int j_ = (unsigned int)(_n);                                 \
00263         (_log2) = (j_ <= 1 ? 0 : 32 - js_bitscan_clz32(j_ - 1));              \
00264     JS_END_MACRO
00265 #else
00266 # define JS_CEILING_LOG2(_log2,_n)                                            \
00267     JS_BEGIN_MACRO                                                            \
00268         uint32_t j_ = (uint32_t)(_n);                                         \
00269         (_log2) = 0;                                                          \
00270         if ((j_) & ((j_)-1))                                                  \
00271             (_log2) += 1;                                                     \
00272         if ((j_) >> 16)                                                       \
00273             (_log2) += 16, (j_) >>= 16;                                       \
00274         if ((j_) >> 8)                                                        \
00275             (_log2) += 8, (j_) >>= 8;                                         \
00276         if ((j_) >> 4)                                                        \
00277             (_log2) += 4, (j_) >>= 4;                                         \
00278         if ((j_) >> 2)                                                        \
00279             (_log2) += 2, (j_) >>= 2;                                         \
00280         if ((j_) >> 1)                                                        \
00281             (_log2) += 1;                                                     \
00282     JS_END_MACRO
00283 #endif

00286 ** Macro version of JS_FloorLog2: Compute the log of the greatest power of
00287 ** 2 less than or equal to _n. The result is returned in _log2.
00288 **
00289 ** This is equivalent to finding the highest set bit in the word.

00291 #ifdef JS_HAS_BUILTIN_BITSCAN32
00292 /*
00293  * Use js_bitscan_clz32 or count-leading-zeros to calculate floor(log2(_n)).

00300     JS_END_MACRO

00303     JS_BEGIN_MACRO                                                            \
00304         uint32_t j_ = (uint32_t)(_n);                                         \
00305         (_log2) = 0;                                                          \
00306         if ((j_) >> 16)                                                       \
00307             (_log2) += 16, (j_) >>= 16;                                       \
00308         if ((j_) >> 8)                                                        \
00309             (_log2) += 8, (j_) >>= 8;                                         \
00310         if ((j_) >> 4)                                                        \
00311             (_log2) += 4, (j_) >>= 4;                                         \
00312         if ((j_) >> 2)                                                        \
00313             (_log2) += 2, (j_) >>= 2;                                         \
00314         if ((j_) >> 1)                                                        \

00383 /*
00384  * User guide to memory management within SpiderMonkey:
00385  *
00386  * Quick tips:
00387  *
00388  *   Allocation:
00389  *   - Prefer to allocate using JSContext:
00390  *       cx->{malloc_,realloc_,calloc_,new_,array_new}
00391  *
00392  *   - If no JSContext is available, use a JSRuntime:
00393  *       rt->{malloc_,realloc_,calloc_,new_,array_new}
00394  *
00395  *   - As a last resort, use unaccounted allocation ("OffTheBooks"):
00396  *       js::OffTheBooks::{malloc_,realloc_,calloc_,new_,array_new}
00397  *
00398  *   Deallocation:
00399  *   - When the deallocation occurs on a slow path, use:
00400  *       Foreground::{free_,delete_,array_delete}
00401  *
00402  *   - Otherwise deallocate on a background thread using a JSContext:
00403  *       cx->{free_,delete_,array_delete}
00404  *
00405  *   - If no JSContext is available, use a JSRuntime:
00406  *       rt->{free_,delete_,array_delete}
00407  *
00408  *   - As a last resort, use UnwantedForeground deallocation:
00409  *       js::UnwantedForeground::{free_,delete_,array_delete}
00410  *
00411  * General tips:
00412  *
00413  *   - Mixing and matching these allocators is allowed (you may free memory
00414  *     allocated by any allocator, with any deallocator).
00415  *
00416  *   - Never, ever use normal C/C++ memory management:
00417  *       malloc, free, new, new[], delete, operator new, etc.
00418  *
00419  *   - Never, ever use low-level SpiderMonkey allocators:
00420  *       js_malloc(), js_free(), js_calloc(), js_realloc()
00421  *     Their use is reserved for the other memory managers.
00422  *
00423  *   - Classes which have private constructors or destructors should have
00424  *     JS_DECLARE_ALLOCATION_FRIENDS_FOR_PRIVATE_CONSTRUCTOR added to their
00425  *     declaration.
00426  *
00427  * Details:
00428  *
00429  *   Using vanilla new/new[] is unsafe in SpiderMonkey because they throw on
00430  *   failure instead of returning NULL, which is what SpiderMonkey expects.
00431  *   (Even overriding them is unsafe, as the system's C++ runtime library may
00432  *   throw, which we do not support. We also can't just use the 'nothrow'
00433  *   variant of new/new[], because we want to mediate *all* allocations
00434  *   within SpiderMonkey, to satisfy any embedders using
00435  *   JS_USE_CUSTOM_ALLOCATOR.)
00436  *
00437  *   JSContexts and JSRuntimes keep track of memory allocated, and use this
00438  *   accounting to schedule GC. OffTheBooks does not. We'd like to remove
00439  *   OffTheBooks allocations as much as possible (bug 636558).
00440  *
00441  *   On allocation failure, a JSContext correctly reports an error, which a
00442  *   JSRuntime and OffTheBooks does not.
00443  *
00444  *   A JSContext deallocates in a background thread. A JSRuntime might
00445  *   deallocate in the background in the future, but does not now. Foreground
00446  *   deallocation is preferable on slow paths. UnwantedForeground deallocations
00447  *   occur where we have no JSContext or JSRuntime, and the deallocation is not
00448  *   on a slow path. We want to remove UnwantedForeground deallocations (bug
00449  *   636561).
00450  *
00451  *   JS_DECLARE_ALLOCATION_FRIENDS_FOR_PRIVATE_CONSTRUCTOR makes the allocation
00452  *   classes friends with your class, giving them access to private
00453  *   constructors and destructors.
00454  *
00455  *   |make check| does a source level check on the number of uses OffTheBooks,
00456  *   UnwantedForeground, js_malloc, js_free etc, to prevent regressions. If you
00457  *   really must add one, update Makefile.in, and run |make check|.
00458  *
00459  *   |make check| also statically prevents the use of vanilla new/new[].
00460  */

00752 /*
00753  * "Move" References
00754  *
00755  * Some types can be copied much more efficiently if we know the original's
00756  * value need not be preserved --- that is, if we are doing a "move", not a
00757  * "copy". For example, if we have:
00758  *
00759  *   Vector<T> u;
00760  *   Vector<T> v(u);
00761  *
00762  * the constructor for v must apply a copy constructor to each element of u ---
00763  * taking time linear in the length of u. However, if we know we will not need u
00764  * any more once v has been initialized, then we could initialize v very
00765  * efficiently simply by stealing u's dynamically allocated buffer and giving it
00766  * to v --- a constant-time operation, regardless of the size of u.
00767  *
00768  * Moves often appear in container implementations. For example, when we append
00769  * to a vector, we may need to resize its buffer. This entails moving each of
00770  * its extant elements from the old, smaller buffer to the new, larger buffer.
00771  * But once the elements have been migrated, we're just going to throw away the
00772  * old buffer; we don't care if they still have their values. So if the vector's
00773  * element type can implement "move" more efficiently than "copy", the vector
00774  * resizing should by all means use a "move" operation. Hash tables also need to
00775  * be resized.
00776  *
00777  * The details of the optimization, and whether it's worth applying, vary from
00778  * one type to the next. And while some constructor calls are moves, many really
00779  * are copies, and can't be optimized this way. So we need:
00780  *
00781  * 1) a way for a particular invocation of a copy constructor to say that it's
00782  *    really a move, and that the value of the original isn't important
00783  *    afterwards (althought it must still be safe to destroy); and
00784  *
00785  * 2) a way for a type (like Vector) to announce that it can be moved more
00786  *    efficiently than it can be copied, and provide an implementation of that
00787  *    move operation.
00788  *
00789  * The Move(T &) function takes a reference to a T, and returns an MoveRef<T>
00790  * referring to the same value; that's 1). An MoveRef<T> is simply a reference
00791  * to a T, annotated to say that a copy constructor applied to it may move that
00792  * T, instead of copying it. Finally, a constructor that accepts an MoveRef<T>
00793  * should perform a more efficient move, instead of a copy, providing 2).
00794  *
00795  * So, where we might define a copy constructor for a class C like this:
00796  *
00797  *   C(const C &rhs) { ... copy rhs to this ... }
00798  *
00799  * we would declare a move constructor like this:
00800  *
00801  *   C(MoveRef<C> rhs) { ... move rhs to this ... }
00802  *
00803  * And where we might perform a copy like this:
00804  *
00805  *   C c2(c1);
00806  *
00807  * we would perform a move like this:
00808  *
00809  *   C c2(Move(c1))
00810  *
00811  * Note that MoveRef<T> implicitly converts to T &, so you can pass an
00812  * MoveRef<T> to an ordinary copy constructor for a type that doesn't support a
00813  * special move constructor, and you'll just get a copy. This means that
00814  * templates can use Move whenever they know they won't use the original value
00815  * any more, even if they're not sure whether the type at hand has a specialized
00816  * move constructor. If it doesn't, the MoveRef<T> will just convert to a T &,
00817  * and the ordinary copy constructor will apply.
00818  *
00819  * A class with a move constructor can also provide a move assignment operator,
00820  * which runs this's destructor, and then applies the move constructor to
00821  * *this's memory. A typical definition:
00822  *
00823  *   C &operator=(MoveRef<C> rhs) {
00824  *     this->~C();
00825  *     new(this) C(rhs);
00826  *     return *this;
00827  *   }
00828  *
00829  * With that in place, one can write move assignments like this:
00830  *
00831  *   c2 = Move(c1);
00832  *
00833  * This destroys c1, moves c1's value to c2, and leaves c1 in an undefined but
00834  * destructible state.
00835  *
00836  * This header file defines MoveRef and Move in the js namespace. It's up to
00837  * individual containers to annotate moves as such, by calling Move; and it's up
00838  * to individual types to define move constructors.
00839  *
00840  * One hint: if you're writing a move constructor where the type has members
00841  * that should be moved themselves, it's much nicer to write this:
00842  *
00843  *   C(MoveRef<C> c) : x(c->x), y(c->y) { }
00844  *
00845  * than the equivalent:
00846  *
00847  *   C(MoveRef<C> c) { new(&x) X(c->x); new(&y) Y(c->y); }
00848  *
00849  * especially since GNU C++ fails to notice that this does indeed initialize x
00850  * and y, which may matter if they're const.
00851  */
00852 template<typename T>
00853 class MoveRef {
00854   public:
00855     typedef T Referent;
00856     explicit MoveRef(T &t) : pointer(&t) { }
00857     T &operator*()  const { return *pointer; }
00858     T *operator->() const { return  pointer; }
00859     operator T& ()   const { return *pointer; }
00860   private:
00861     T *pointer;
00862 };
00863 
00864 template<typename T>
00865 MoveRef<T> Move(T &t) { return MoveRef<T>(t); }
00866 
00867 template<typename T>
00868 MoveRef<T> Move(const T &t) { return MoveRef<T>(const_cast<T &>(t)); }

00918 /*
00919  * Given a raw hash code, h, return a number that can be used to select a hash
00920  * bucket.
00921  *
00922  * This function aims to produce as uniform an output distribution as possible,
00923  * especially in the most significant (leftmost) bits, even though the input
00924  * distribution may be highly nonrandom, given the constraints that this must
00925  * be deterministic and quick to compute.
00926  *
00927  * Since the leftmost bits of the result are best, the hash bucket index is
00928  * computed by doing ScrambleHashCode(h) / (2^32/N) or the equivalent
00929  * right-shift, not ScrambleHashCode(h) % N or the equivalent bit-mask.
00930  *
00931  * FIXME: OrderedHashTable uses a bit-mask; see bug 775896.
00932  */
00933 inline HashNumber
00934 ScrambleHashCode(HashNumber h)
00935 {
00936     /*
00937      * Simply returning h would not cause any hash tables to produce wrong
00938      * answers. But it can produce pathologically bad performance: The caller
00939      * right-shifts the result, keeping only the highest bits. The high bits of
00940      * hash codes are very often completely entropy-free. (So are the lowest
00941      * bits.)
00942      *
00943      * So we use Fibonacci hashing, as described in Knuth, The Art of Computer
00944      * Programming, 6.4. This mixes all the bits of the input hash code h.
00945      * 
00946      * The value of goldenRatio is taken from the hex
00947      * expansion of the golden ratio, which starts 1.9E3779B9....
00948      * This value is especially good if values with consecutive hash codes
00949      * are stored in a hash table; see Knuth for details.
00950      */
00951     static const HashNumber goldenRatio = 0x9E3779B9U;
00952     return h * goldenRatio;
00953 }
